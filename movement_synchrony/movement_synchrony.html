def game_period(vmrk,len_of_mot):
    """ Parses vmrk, obtains start trigger, reads eyes open and closed triggers and returns 
        trial times as boolen array
        Args:
            vmrk: Path to vmrk file
            len_of_mot: length of motion file
        Returns:
            Game period as boolean array. True for game period.        
    """
    vmrk = helper.parse_vmrk(vmrk)
    video_start_trigger = np.array(vmrk["time"], dtype = int)[np.array(vmrk['description'])=='S128'][0]

    eyes_open = np.array(vmrk['description'])=='S 21'
    eyes_open = np.array(vmrk["time"], dtype = np.int32)[eyes_open]
    eyes_open = (eyes_open - video_start_trigger)//20

    eyes_closed = np.array(vmrk['description'])=='S 20'
    eyes_closed = np.array(vmrk["time"], dtype = np.int32)[eyes_closed]
    eyes_closed = (eyes_closed - video_start_trigger)//20

    game = np.zeros(len_of_mot,dtype = np.bool)
    game[eyes_open[0]:eyes_closed[1]] = 1
    game[eyes_open[1]:eyes_closed[2]] = 1
    return game
